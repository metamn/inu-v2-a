  9 const propTypesForSemanticHTMLElement = {
 10   /**
 11    * The element name
 12    * Like 'section', 'article', 'aside', 'nav'
 13    */
 14   elementName: PropTypes.string.isRequired,
 15   /**
 16    * The element class name
 17    * Required to make it later styleable
 18    */
 19   className: PropTypes.string.isRequired,
 20   /**
 21    * The element title
 22    * Usually a string, but it can be a link too
 23    */
 24   title: PropTypes.any.isRequired,
 25   /**
 26    * The element children
 27    * Without children there is no use of an empty element
 28    */
 29   children: PropTypes.any.isRequired
 30 };

 91 /**
 92  * Styles the semantic element title
 93  * - It is always hidden because it is needed only by the HTML outliner / validator
 94  * @type {[type]}
 95  */
 96 const Title = styled("h3")([], {
 97   display: "none"
 98 });
 99 
100 /**
101  * Creates a semantic HTML element with title
102  * - Semantic elements with title are properly outlined in https://validator.w3.org/
103  * - When a HTML document outlines perfectly it means its component structure is flawless
104  * - Many times an invalid outline structure points to errors in component design and helps fix it
105  *
106  * Returns something like: `<section><h3>section title</h3>...children</section>` where the title is hidden by default
107  *
108  * NOTE: since we have inside an element a title + content it will act as a list so we have to provide unique `key` props
109  *
110  * @param {[type]} props [description]
111  */
112 const SemanticHTMLElement = props => {
113   const { elementName, children, className, title } = props;
114 
115   const titleElement = React.createElement(
116     Title,
117     { className: "title", key: 1 },
118     title
119   );
120 
121   return React.createElement(elementName, { className: className, key: 2 }, [
122     titleElement,
123     children
124   ]);
125 };
126 
127 SemanticHTMLElement.propTypes = propTypesForSemanticHTMLElement;
128 
129 /**
130  * Creates a `<section>` element
131  * @param {[type]} props [description]
132  */
133 const Section = props => {
134   return SemanticHTMLElement(props);
135 };
