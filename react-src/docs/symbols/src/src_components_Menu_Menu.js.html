1 import React, { useState } from "react";
  2 import PropTypes from "prop-types";
  3 import styled from "styled-components";
  4 
  5 import { useLocalStorage } from "./../../hooks";
  6 
  7 import Categories from "../Categories";
  8 import Category, { setCategoryStatus } from "../Category";
  9 import { Nav as _Nav } from "../SemanticHTML";
 10 
 11 /**
 12  * Defines the prop types
 13  */
 14 const propTypes = {
 15   /**
 16    * Defines prop types for the Random extra menu item
 17    */
 18   random: Category.propTypes,
 19   /**
 20    * Defines prop types for the Contact extra menu item
 21    */
 22   contact: Category.propTypes
 23 };
 24 
 25 /**
 26  * Defines the default props
 27  */
 28 const defaultProps = {
 29   /**
 30    * Defines default props for the Random extra menu item
 31    */
 32   random: {
 33     id: "random",
 34     categoryId: "-1",
 35     name: "Random"
 36   },
 37   /**
 38    * Defines default props for the Contact extra menu item
 39    */
 40   contact: {
 41     id: "contact",
 42     categoryId: "-2",
 43     name: "Contact"
 44   }
 45 };
 46 
 47 /**
 48  * Displays the Menu container
 49  */
 50 const Nav = styled(_Nav)(props => ({}));
 51 
 52 /**
 53  * Displays the the extra menu items container
 54  */
 55 const ExtraMenuItems = styled("ul")(props => ({
 56   marginTop: "var(--lem)"
 57 }));
 58 
 59 /**
 60  * Displays the Menu
 61  * @param Object props The component properties
 62  */
 63 const Menu = props => {
 64   /**
 65    * Saves the active category into a state
 66    */
 67   const [activeCategory, setActiveCategory] = useState(1);
 68 
 69   /**

 71    * At the next load exactly the same number of temporary categories will be displayed as the real number of categories
 72    */
 73   const [numberOfEdgesSaved, setNumberOfEdgesSaved] = useLocalStorage(
 74     "number-of-edges"
 75   );
 76 
 77   return (
 78     <Nav title="Menu" className="menu">
 79       <Categories
 80         activeCategory={activeCategory}
 81         setActiveCategory={setActiveCategory}

 84       />

 86       <ExtraMenuItems className="extra-menu-items">
 87         <Category
 88           {...defaultProps.random}
 89           status={setCategoryStatus({

 92           })}
 93           setActiveCategory={setActiveCategory}
 94         />
 95         <Category
 96           {...defaultProps.contact}
 97           status={setCategoryStatus({
 98             categoryId: defaultProps.contact.categoryId,
 99             activeCategory: activeCategory
