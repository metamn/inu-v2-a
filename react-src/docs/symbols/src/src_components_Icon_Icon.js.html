1 import React from "react";
  2 import PropTypes from "prop-types";
  3 import styled from "styled-components";
  4 
  5 import { useTheme } from "../../hooks";
  6 
  7 /**
  8  * Defines the prop types of the component
  9  * @type {Object}
 10  */
 11 const propTypes = {
 12   /**
 13    * The size multiplier
 14    * The width, height of the icon will be `var(--lem) * size`
 15    * @type {[type]}
 16    */
 17   size: PropTypes.number,
 18   /**
 19    * The icon status
 20    * @type {[type]}
 21    */
 22   status: PropTypes.oneOf(["active", "inactive", "hidden"]),
 23   /**
 24    * The icon itself. Preferably in SVG format.
 25    * @type {[type]}
 26    */
 27   children: PropTypes.any.isRequired
 28 };
 29 
 30 /**
 31  * Defines the default props
 32  * @type {Object}
 33  */
 34 const defaultProps = {
 35   size: 1.5,
 36   status: "active",
 37   children: ""
 38 };
 39 
 40 /**
 41  * Displays the icon container
 42  * @type {[type]}
 43  */
 44 const Container = styled("div")(props => ({
 45   width: `calc(var(--lem) * ${props.size})`,
 46   height: `calc(var(--lem) * ${props.size})`,
 47 
 48   visibility: props.status === "hidden" ? "hidden" : "visible",
 49   color:
 50     props.status === "active"
 51       ? props.theme.colors.text
 52       : props.theme.colors.inactive,
 53   cursor: props.status === "active" ? "pointer" : "default",
 54 
 55   svg: {
 56     fontSize: `calc(var(--lem) * ${props.size})`
 57   }
 58 }));
 59 
 60 /**
 61  * Displays an icon
 62  * @param Object props The component properties
 63  */

 66   const { theme } = useTheme();
 67   //console.log("icon");
 68 
 69   return (
 70     <Container className="icon" theme={theme} {...props}>
 71       {children}
