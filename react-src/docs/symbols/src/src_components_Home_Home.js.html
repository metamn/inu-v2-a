1 import React from "react";
  2 import styled from "styled-components";
  3 import gql from "graphql-tag";
  4 import WebFont from "webfontloader";
  5 //import { stringify } from "flatted";
  6 
  7 import { useTheme, useData } from "./../../hooks";
  8 
  9 import Reset from "../Reset";
 10 import TypographicGrid from "../TypographicGrid";
 11 
 12 import SiteInfo from "../SiteInfo";
 13 import { Section as _Section } from "../SemanticHTML";
 14 import Header from "../Header";
 15 
 16 /**
 17  * Loads web fonts
 18  * NOTE: When changing font also the `theme` has to be updated
 19  */
 20 WebFont.load({
 21   google: {
 22     families: ["Major+Mono+Display"]
 23   }
 24 });
 25 
 26 /**
 27  * Defines the prop types of the component
 28  * @type Object
 29  */
 30 const propTypes = {
 31   ...SiteInfo.propTypes
 32 };
 33 
 34 /**
 35  * Defines the default props
 36  * @type Object
 37  */
 38 const defaultProps = {
 39   ...SiteInfo.defaultProps
 40 };
 41 
 42 /**
 43  * Defines the query
 44  * @type {[type]}
 45  */
 46 const query = gql`
 47   query siteInfo {
 48     generalSettings {
 49       ...SiteInfoSettings
 50     }
 51   }
 52   ${SiteInfo.fragments.settings}
 53 `;
 54 
 55 /**
 56  * Styles the main container
 57  * Wherever possible the object notation is used to enable composition by spreading and reduce the ugly, long `{$props => ...}` syntax with template literals
 58  *
 59  * @type {[type]}
 60  */
 61 const Section = styled(_Section)(props => ({
 62   ...props.theme.colorPairs.default,
 63   ...props.theme.fonts.default,
 64   display: "flex",
 65   flexDirection: "column",
 66   minHeight: "100vh"
 67 }));
 68 
 69 /**
 70  * Displays the Homepage
 71  * @param Object props The component properties
 72  */
 73 const Home = props => {
 74   // Theming
 75   const { currentTheme, switchTheme, ThemeContext } = useTheme();
 76   //console.log("home");
 77 
 78   // Site info
 79   const siteInfo = useData(props, query, "generalSettings");
 80 
 81   return (
 82     <>
 83       <Reset />
 84       <SiteInfo {...siteInfo} />
 85       <ThemeContext.Provider value={currentTheme}>
 86         <TypographicGrid />
 87         <Section title="Home" className="home" theme={currentTheme.theme}>
 88           <Header {...siteInfo} themeIconClickHandler={switchTheme} />
 89         </Section>
 90       </ThemeContext.Provider>
 91     </>
 92   );
 93 };
 94 
 95 Home.propTypes = propTypes;
 96 Home.defaultProps = defaultProps;
 97 
 98 export default Home;
