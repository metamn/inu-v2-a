1 import React from "react";
  2 import PropTypes from "prop-types";
  3 import gql from "graphql-tag";
  4 import { Helmet } from "react-helmet";
  5 
  6 /**
  7  * Defines the prop types of the component
  8  * @type Object
  9  */
 10 const propTypes = {
 11   /**
 12    * The site title
 13    */
 14   title: PropTypes.string,
 15   /**
 16    * The site description
 17    */
 18   description: PropTypes.string,
 19   /**
 20    * The site url
 21    */
 22   url: PropTypes.string
 23 };
 24 
 25 /**
 26  * Defines the default props
 27  * @type Object
 28  */
 29 const defaultProps = {
 30   title: "Ioan Chivu",
 31   description: "Photo traveler",
 32   url: "http://inu.ro"
 33 };
 34 
 35 /**
 36  * Defines the query fragment needed by the component
 37  * @type Object
 38  */
 39 const queryFragment = {
 40   settings: gql`
 41     fragment SiteInfoSettings on GeneralSettings {
 42       title
 43       description
 44       url
 45     }
 46   `
 47 };
 48 
 49 /**
 50  * Displays site info in the document `<head>` with Helmet
 51  * NOTE: use this method instead of manually edit `public/index.php`
 52  * @param Object props The component properties
 53  */
 54 const SiteInfo = props => {
 55   const { title, description, url } = props;
 56 
 57   return (
 58     <Helmet>
 59       <title>{title}</title>
 60       <meta name="description" content={description} />
 61       <link rel="canonical" href={url} />
 62     </Helmet>
 63   );
 64 };
 65 
 66 SiteInfo.propTypes = propTypes;
 67 SiteInfo.defaultProps = defaultProps;
 68 SiteInfo.fragments = queryFragment;
 69 
 70 export default SiteInfo;
