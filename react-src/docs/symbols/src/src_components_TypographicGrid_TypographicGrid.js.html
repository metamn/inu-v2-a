1 import React, { useContext } from "react";
  2 import PropTypes from "prop-types";
  3 import styled, { createGlobalStyle } from "styled-components";
  4 import Repeat from "../Repeat";
  5 
  6 import { ThemeContext } from "../../themes/default.js";
  7 
  8 /**
  9  * Defines the component prop types
 10  * @type Object
 11  */
 12 const propTypes = {
 13   /**
 14    * Display vertical rhythm lines?
 15    */
 16   displayVerticalRhytm: PropTypes.bool,
 17   /**
 18    * Display horizontal rhythm lines?
 19    */
 20   displayHorizontalRhytm: PropTypes.bool,
 21   /**
 22    * The number of horizontal rhythm lines
 23    */
 24   numberOfHorizontalLines: PropTypes.number,
 25   /**
 26    * The number of vertical rhythm lines
 27    */
 28   numberOfVerticalLines: PropTypes.number,
 29   /**
 30    * The color of the rhythm lines
 31    */
 32   lineColor: PropTypes.string
 33 };
 34 
 35 /**
 36  * Defines the default props
 37  * @type Object
 38  */
 39 const defaultProps = {
 40   displayVerticalRhytm: false,
 41   displayHorizontalRhytm: false,
 42   numberOfHorizontalLines: 100,
 43   numberOfVerticalLines: 100,
 44   lineColor: "gray"
 45 };
 46 
 47 /**
 48  * Defines the component name
 49  * @type String
 50  */
 51 const className = "typographic-grid";
 52 
 53 /**
 54  * Sets up the typographic grid
 55  * It's hand coded since `createGlobalStyle` doesn't supports yet object notation
 56  * @type String
 57  */
 58 const GlobalStyle = createGlobalStyle`
 59 	body {
 60 		font-size: 100%;
 61 	    line-height: 1.25;
 62 	    --lem: 1.25em;
 63 	}
 64 `;
 65 
 66 /**
 67  * Shared settings for the rhythm containers
 68  * @type String
 69  */
 70 const Rhythm = styled("div")([], {
 71   position: "absolute",
 72   top: 0,
 73   left: 0,
 74   width: "100%",
 75   height: "100%"
 76 });
 77 
 78 /**
 79  * The vertical rhythm container
 80  * @type String
 81  */
 82 const VerticalRhythm = styled(Rhythm)(props => ({
 83   display: props.displayVerticalRhytm ? "flex" : "none",
 84   flexWrap: "wrap"
 85 }));
 86 
 87 /**
 88  * The horizontal rhythm container
 89  * @type String
 90  */
 91 const HorizontalRhythm = styled(Rhythm)(props => ({
 92   display: props.displayHorizontalRhytm ? "flex" : "none",
 93   flexWrap: "wrap"
 94 }));
 95 
 96 /**
 97  * Shared settings for the rhythm lines
 98  * @type String
 99  */
100 const Line = styled.div(props => ({
101   boxSizing: "border-box",
102   borderColor: props.lineColor ? props.lineColor : "transparent",
103   borderWidth: "1px"
104 }));
105 
106 /**
107  * The vertical rhythm line
108  * @type String
109  */
110 const VerticalRhythmLine = styled(Line)([], {
111   width: "var(--lem)",
112   height: "100%",
113   borderRightStyle: "solid"
114 });
115 
116 /**
117  * The horizontal rhythm line
118  * @type String
119  */
120 const HorizontalRhythmLine = styled(Line)([], {
121   width: "100%",
122   height: "var(--lem)",
123   borderBottomStyle: "solid"
124 });
125 
126 /**
127  * The main container
128  * @type Sting
129  */
130 const Container = styled.div([]);
131 
132 /**
133  * Displays horizontal and vertical lines across the page
134  * It helps to verify elements are all perfectly aligned to the grid
135  * See: http://metamn.io/mr-ui
136  * @param Object props The component properties
137  */
138 const TypographicGrid = props => {
139   const {
140     displayVerticalRhytm,
141     displayHorizontalRhytm,
142     numberOfHorizontalLines,
143     numberOfVerticalLines,
144     lineColor
145   } = props;
146 
147   const themeContext = useContext(ThemeContext);
148   const { theme } = themeContext;
149 
150   return (
151     <>
152       <GlobalStyle theme={theme} />
153       {(displayHorizontalRhytm || displayVerticalRhytm) && (
154         <Container className={className}>
155           {displayHorizontalRhytm && (
156             <HorizontalRhythm
157               className="horizontal-lines"
158               displayHorizontalRhytm={displayHorizontalRhytm}
159             >
160               <Repeat numberOfTimes={numberOfHorizontalLines} startAt={0}>
161                 {i => <HorizontalRhythmLine key={i} lineColor={lineColor} />}
162               </Repeat>
163             </HorizontalRhythm>
164           )}
165           {displayVerticalRhytm && (
166             <VerticalRhythm
167               className="vertical-lines"
168               displayVerticalRhytm={displayVerticalRhytm}
169             >
170               <Repeat numberOfTimes={numberOfVerticalLines} startAt={0}>
171                 {i => <VerticalRhythmLine key={i} lineColor={lineColor} />}
172               </Repeat>
173             </VerticalRhythm>
174           )}
175         </Container>
176       )}
177     <
