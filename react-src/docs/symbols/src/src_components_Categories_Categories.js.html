1 import React from "react";
  2 import PropTypes from "prop-types";
  3 import styled from "styled-components";
  4 import gql from "graphql-tag";
  5 import { stringify } from "flatted";
  6 
  7 import { useTheme, useData } from "./../../hooks";
  8 import Category, { setCategoryStatus } from "../Category";
  9 
 10 /**
 11  * Defines the prop types
 12  */
 13 const propTypes = {
 14   /**
 15    * Defines a category node
 16    */
 17   node: Category.propTypes,
 18   /**
 19    * Defines a list of category nodes
 20    */
 21   edges: PropTypes.array,
 22   /**
 23    * Defines the number of edges
 24    */
 25   numberOfEdges: PropTypes.number,
 26   /**
 27    * Defines the number of edges saved to local storage
 28    */
 29   numberOfEdgesSaved: PropTypes.number,
 30   /**
 31    * Defines the function which sets the number of edges saved to local storage
 32    */
 33   setNumberOfEdgesSaved: PropTypes.func,
 34   /**
 35    * Defines the active category
 36    */
 37   activeCategory: PropTypes.number,
 38   /**
 39    * Defines the function which sets the active category
 40    */
 41   setActiveCategory: PropTypes.func
 42 };
 43 
 44 /**
 45  * Defines the default props
 46  */
 47 const defaultProps = {
 48   node: Category.defaultProps,
 49   edges: [],
 50   numberOfEdges: 5,
 51   setNumberOfEdgesSaved: () => {},
 52   activeCategory: 1,
 53   setActiveCategory: () => {}
 54 };
 55 
 56 /**
 57  * Defines the query
 58  */
 59 const query = gql`
 60   query Categories($hideEmpty: Boolean) {
 61     categories(where: { hideEmpty: $hideEmpty, orderby: TERM_ORDER }) {
 62       edges {
 63         node {
 64           ...CategoryNode
 65         }
 66       }
 67     }
 68   }
 69   ${Category.fragments.node}
 70 `;
 71 
 72 /**
 73  * Displays the Categories container
 74  */
 75 const Container = styled("ul")(props => ({}));
 76 
 77 /**
 78  * Creates temporary data to be displayed while real data is loading
 79  *
 80  * @param  {Object} props The Object from the temporary data will be created
 81  * @return {Object}       An array of data
 82  */
 83 const createTemporaryData = props => {
 84   const { node, numberOfEdges, numberOfEdgesSaved } = props;
 85 
 86   /**
 87    * Defines the number of edges to be displayed either from default props or from the local storage
 88    */
 89   const newNumberOfEdges =
 90     typeof numberOfEdgesSaved !== "undefined"
 91       ? numberOfEdgesSaved
 92       : numberOfEdges;
 93 
 94   return { edges: Array(newNumberOfEdges).fill({ node: node }) };
 95 };
 96 
 97 /**
 98  * Displays the Categories
 99  * @param Object props The component properties
100  */
101 const Categories = props => {
102   const {
103     activeCategory,
104     setActiveCategory,
105     numberOfEdgesSaved,
106     setNumberOfEdgesSaved
107   } = props;
108 
109   /**
110    * Loads temporary data
111    */
112   const tempData = createTemporaryData({ numberOfEdgesSaved, ...defaultProps });
113 
114   /**
115    * Loads the real data
116    */
117   const data = useData(tempData, query, "categories");
118 
119   /**
120    * Displays the data
121    */
122   const items = data.edges.map(edge => (
123     <Category
124       {...edge.node}
125       status={setCategoryStatus({
126         categoryId: edge.node.categoryId,
127         activeCategory: activeCategory
128       })}
129       setActiveCategory={setActiveCategory}
130     />
131   ));
132 
133   /**
134    * Saves the number of categories to local storage
135    */
136   setNumberOfEdgesSaved(data.edges.length);
137 
138   /**
139    * Marks the first category as active
140    */
141   setActiveCategory(data.edges[0].node.categoryId);
142 
143   //console.log("category");
144 
145   return <Container className="categories">{items}</Container>;
146 };
147 
148 Categories.propTypes = propTypes;
149 Categories.defaultProps = defaultProps;
150 
151 export default Categories;
