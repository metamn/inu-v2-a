1 /**
  2  * Imitates the for loop inside JSX
  3  * - https://reactjs.org/docs/jsx-in-depth.html
  4  * @param Object props The component properties
  5  */
  6 const Repeat = props => {
  7   let ret = [];
  8   let startAt = props.startAt;
  9   let endAt = startAt + props.numberOfTimes;
 10 
 11   for (let i = startAt; i < endAt; i++) {
 12     ret.push(props.children(i));
 13   }
 14 
 15   return ret;
 16 };
 17 
 18 export default Repeat;
